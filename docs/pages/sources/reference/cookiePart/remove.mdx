import {Badge} from '../../../../';

# remove <Badge type={'method'}>Method</Badge>

Removes the current value

## API

```ts copy
remove(): void;
```

## Example

```ts copy
import {cookiePart} from '@bitfiber/rx';

const now = new Date();
const token = cookiePart('token', {
  path: '/some-path',
  domain: '.example.com',
  expires: new Date(now.getTime() - 1000*60*60*24),
  maxAge: -1,
  secure: true,
  sameSite: 'strict',
});

token.remove();
```
