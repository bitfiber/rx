# $ <Badge type={'property'}>Property</Badge> <Badge type={'modifier'}>Readonly</Badge>

Allows subscribers to reactively observe value changes

## API

```ts copy
readonly $: Observable<T>;
```

## Example

```ts copy
import {map} from 'rxjs';
import {KeyValueSourcePart, MemoryStorage} from '@bitfiber/rx';

// Creates a memory source part
const source = new KeyValueSourcePart<number>('someKey', new MemoryStorage());

// Uses the source observable
source.$
  .pipe(map(v => v * 2))
  .subscribe(v => console.log(v));
```
